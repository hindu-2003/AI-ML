#Drawing Single Sub Plot
import matplotlib.pyplot as plt

# function to create a figure
fig = plt.figure(figsize=(10,6))
# adding subplot to the created figure
fig1= fig.add_subplot(111)
x = [1,2,3,6,7,8]
y = [3,4,6,1,3,0]  
plt.xlabel('x- axis')
plt.ylabel('y-axis')
plt.title('Adding plot on a figure')
# plotting points on the subplot
fig1.plot(x, y)
plt.grid()
plt.show()

#Code Sine wave and Cos wave
import matplotlib.pyplot as plt
import numpy as np
fig = plt.figure(figsize=(10,8),facecolor='lightblue')
x = np.arange(-np.pi,np.pi,0.01)

fig1 = fig.add_subplot(2,1,1)
fig1.plot(x,np.sin(x))
plt.title('Sine curve')

fig2 = fig.add_subplot(2,1,2)
fig2.plot(x,np.cos(x))
plt.title('Cosine curve')

fig.suptitle('Subplots in a figure',fontsize=15)
plt.show()

import matplotlib.pyplot as plt
import numpy as np

fig = plt.figure(figsize=(12, 6))

x = np.arange(0, 10, 0.1)
y = np.sin(x)
z = np.cos(x)

ax1 = fig.add_subplot(121)
ax2 = fig.add_subplot(122)

ax1.set_title('Full view')
ax1.plot(y, color='blue', label='Sine wave')
ax1.plot(z, color='black', label='Cosine wave')

ax2.set_title('Truncated view')
ax2.plot(y, color='blue', label='Sine wave')
ax2.plot(z, color='black', label='Cosine wave')

ax2.set_xlim([25, 50])

plt.show()
